## 3️⃣ 規則引擎 Service 實作

```java
@Service
@RequiredArgsConstructor
@Slf4j
public class EventRuleService {
    
    private final EventRuleRepository ruleRepository;
    
    /**
     * 評估 telemetry 數據是否觸發任何規則
     */
    public List<EventRule> evaluateRules(String deviceType, Map<String, Object> telemetry) {
        List<EventRule> rules = ruleRepository.findByDeviceTypeAndEnabledTrue(deviceType);
        
        return rules.stream()
                .filter(rule -> evaluateCondition(rule.getCondition(), telemetry))
                .toList();
    }
    
    /**
     * 遞迴評估條件（支援巢狀 AND/OR）
     */
    private boolean evaluateCondition(Map<String, Object> condition, Map<String, Object> data) {
        String operator = (String) condition.get("operator"); // "AND" or "OR"
        List<Object> conditions = (List<Object>) condition.get("conditions");
        
        if ("AND".equals(operator)) {
            return conditions.stream()
                    .allMatch(c -> evaluateSingleCondition((Map<String, Object>) c, data));
        } else if ("OR".equals(operator)) {
            return conditions.stream()
                    .anyMatch(c -> evaluateSingleCondition((Map<String, Object>) c, data));
        }
        
        return false;
    }
    
    /**
     * 評估單個條件（可能也是巢狀條件）
     */
    private boolean evaluateSingleCondition(Map<String, Object> condition, Map<String, Object> data) {
        // 如果是巢狀條件（有 operator 和 conditions）
        if (condition.containsKey("operator") && condition.containsKey("conditions")) {
            return evaluateCondition(condition, data);
        }
        
        // 單欄位條件
        String field = (String) condition.get("field");
        String operator = (String) condition.get("operator");
        Object expectedValue = condition.get("value");
        
        Object actualValue = data.get(field);
        if (actualValue == null) return false;
        
        return compareValues(actualValue, operator, expectedValue);
    }
    
    /**
     * 數值比較
     */
    private boolean compareValues(Object actual, String operator, Object expected) {
        if (!(actual instanceof Number) || !(expected instanceof Number)) {
            // 字串比較
            return compareStrings(String.valueOf(actual), operator, String.valueOf(expected));
        }
        
        double actualVal = ((Number) actual).doubleValue();
        double expectedVal = ((Number) expected).doubleValue();
        
        return switch (operator) {
            case ">" -> actualVal > expectedVal;
            case ">=" -> actualVal >= expectedVal;
            case "<" -> actualVal < expectedVal;
            case "<=" -> actualVal <= expectedVal;
            case "==" -> actualVal == expectedVal;
            case "!=" -> actualVal != expectedVal;
            default -> false;
        };
    }
    
    private boolean compareStrings(String actual, String operator, String expected) {
        return switch (operator) {
            case "==" -> actual.equals(expected);
            case "!=" -> !actual.equals(expected);
            case "contains" -> actual.contains(expected);
            case "startsWith" -> actual.startsWith(expected);
            case "endsWith" -> actual.endsWith(expected);
            default -> false;
        };
    }
}
```

```java
// 當收到路燈 telemetry 時
public void handleTelemetry(String deviceType, Map<String, Object> telemetry) {
    
    // 1. 驗證數據格式（使用現有的 DeviceTemplate）
    deviceTemplateService.validateAttributes(deviceType, telemetry);
    
    // 2. 評估是否觸發事件規則
    List<EventRule> triggeredRules = eventRuleService.evaluateRules(deviceType, telemetry);
    
    // 3. 處理觸發的事件
    for (EventRule rule : triggeredRules) {
        log.warn("觸發規則: {} - {}", rule.getRuleName(), telemetry);
        
        // 4. 執行動作（發送告警、記錄事件等）
        eventActionService.executeActions(rule.getActions(), telemetry);
    }
    
    // 5. 儲存 telemetry
    telemetryRepository.save(telemetry);
}
```
