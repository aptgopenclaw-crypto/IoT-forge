#### **選項 B：自製遞迴組件（完全客製化）**

如果需要更靈活的 UI，可以自製：

```jsx
// ConditionBuilder.jsx
const ConditionBuilder = ({ value, onChange, fields }) => {
  const addCondition = () => {
    onChange({
      ...value,
      rules: [...(value.rules || []), { field: '', operator: '=', value: '' }]
    });
  };

  const addGroup = () => {
    onChange({
      ...value,
      rules: [...(value.rules || []), { combinator: 'and', rules: [] }]
    });
  };

  const updateRule = (index, newRule) => {
    const newRules = [...value.rules];
    newRules[index] = newRule;
    onChange({ ...value, rules: newRules });
  };

  const removeRule = (index) => {
    const newRules = value.rules.filter((_, i) => i !== index);
    onChange({ ...value, rules: newRules });
  };

  return (
    <div className="condition-builder">
      <Select 
        value={value.combinator || 'and'}
        onChange={v => onChange({ ...value, combinator: v })}
        options={[
          { label: 'AND (全部滿足)', value: 'and' },
          { label: 'OR (任一滿足)', value: 'or' }
        ]}
      />

      <div className="rules-container">
        {value.rules?.map((rule, index) => (
          <div key={index} className="rule-item">
            {rule.combinator ? (
              // 巢狀群組
              <ConditionBuilder 
                value={rule}
                onChange={newRule => updateRule(index, newRule)}
                fields={fields}
              />
            ) : (
              // 單個條件
              <div className="condition-row">
                <Select 
                  value={rule.field}
                  onChange={v => updateRule(index, { ...rule, field: v })}
                  options={fields.map(f => ({ label: f.label, value: f.name }))}
                  placeholder="選擇欄位"
                />
                
                <Select 
                  value={rule.operator}
                  onChange={v => updateRule(index, { ...rule, operator: v })}
                  options={getOperatorsForField(rule.field)}
                />
                
                <Input
                  value={rule.value}
                  onChange={e => updateRule(index, { ...rule, value: e.target.value })}
                  placeholder="輸入值"
                />
                
                <Button onClick={() => removeRule(index)} danger>刪除</Button>
              </div>
            )}
          </div>
        ))}
      </div>

      <div className="actions">
        <Button onClick={addCondition}>+ 添加條件</Button>
        <Button onClick={addGroup}>+ 添加條件組</Button>
      </div>
    </div>
  );
};
```

```jsx
import { Form, Input, Select, Switch, Button, Card, Divider, message } from 'antd';
import QueryBuilder from 'react-querybuilder';
import 'react-querybuilder/dist/query-builder.css';

const EventRuleForm = ({ rule, onSave }) => {
  const [form] = Form.useForm();
  const [query, setQuery] = useState(rule?.condition || { combinator: 'and', rules: [] });
  const [actions, setActions] = useState(rule?.actions || {});
  const [deviceFields, setDeviceFields] = useState([]);

  // 載入設備欄位（從 DeviceTemplate）
  const loadDeviceFields = async (deviceType) => {
    const schema = await api.getDeviceTemplate(deviceType);
    const fields = schema?.fields?.map(f => ({
      name: f.key,
      label: f.title || f.key,
      type: f.type,
      ...f
    })) || [];
    setDeviceFields(fields);
  };

  const handleSubmit = async () => {
    const values = await form.validateFields();
    const payload = {
      ...values,
      condition: query,
      actions
    };
    await onSave(payload);
    message.success('儲存成功');
  };

  return (
    <Card title="規則配置">
      <Form form={form} layout="vertical" initialValues={rule}>
        {/* 基本資訊 */}
        <Form.Item name="ruleName" label="規則名稱" rules={[{ required: true }]}>
          <Input placeholder="輸入規則名稱" />
        </Form.Item>

        <Form.Item name="deviceType" label="設備類型" rules={[{ required: true }]}>
          <Select 
            onChange={loadDeviceFields}
            options={deviceTypes}
          />
        </Form.Item>

        <Divider>條件配置</Divider>
        
        {/* 規則構建器 */}
        <div style={{ marginBottom: 16 }}>
          <QueryBuilder
            query={query}
            onQueryChange={setQuery}
            fields={deviceFields}
            operators={[
              { name: '>', label: '大於' },
              { name: '>=', label: '大於等於' },
              { name: '<', label: '小於' },
              { name: '<=', label: '小於等於' },
              { name: '=', label: '等於' },
              { name: '!=', label: '不等於' }
            ]}
          />
        </div>

        <Divider>動作配置</Divider>
        
        {/* 動作配置 */}
        <Form.Item label="觸發動作">
          <Checkbox.Group value={Object.keys(actions)} onChange={setActions}>
            <Checkbox value="send_alert">發送告警</Checkbox>
            <Checkbox value="log_event">記錄日誌</Checkbox>
          </Checkbox.Group>
        </Form.Item>

        <Form.Item>
          <Button type="primary" onClick={handleSubmit}>儲存規則</Button>
          <Button style={{ marginLeft: 8 }}>取消</Button>
          <Button style={{ marginLeft: 8 }} onClick={testRule}>測試規則</Button>
        </Form.Item>
      </Form>
    </Card>
  );
```
